program fibRecursive;

function int fibo(var int fiboLimite){

	var int fiboAnt;
	var int	fiboAntAnt;

	if(fiboLimite == 0)
	{
		return 0;
	} 
	else
	{
		if(fiboLimite == 1){
			return 1;
		} else
		{

			fiboAnt = fibo(fiboLimite - 1);
			fiboAntAnt = fibo(fiboLimite - 2);

			return fiboAnt + fiboAntAnt;
		}
	}
}

main {
	var int resultado = 0;
	var int fiboLimite = 10;

	resultado = fibo(fiboLimite);
	write(resultado);
}
